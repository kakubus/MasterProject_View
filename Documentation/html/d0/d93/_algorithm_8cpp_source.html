667 oldSolution.calculateTotalCost();
668 newSolution.calculateTotalCost();
669

674 }
675 }

678 char s = ' ';

681 }

683 char s = ' ';

690 return NON_PROFIT;
691 }

693 return UNDEFINED_SOLUTION_ERROR;

816 modifiedSolution->locationsInSolution[v1].erase(modifiedSolution->locationsInSolution[v1].begin() + inV); // Remove location from source vehicle
817 lastChange.log(

819 l1,

821 v1,
822 path.first,
823 modificationType);

828 modifiedSolution->calculateTotalCost();

830 std::cout << "(TS modSolution) - CORRUPTED SOLUTION\n" << std::endl;
831 continue;
