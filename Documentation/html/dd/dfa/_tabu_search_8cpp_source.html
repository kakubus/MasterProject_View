290bool TabuSearch::checkTabu(MethodResult& lastModification)
291{
292 if (_tabuList.size() > 0)
293 {
294 for (auto& tabuElement : _tabuList) {
295 if (lastModification == tabuElement.first) // If found forbidden modification
296 {
297 if (tabuElement.second == 0) // And if forbidden modification is allowed (iterations > tabu tenure)
298 {
299 return true; // Return true. The action being checked is allowed.
300 }
301 else {
302 return false; // Return false. Action is forbidden because tabu period is not expired.
303 }
304 }
305 }

411 std::pair<Vehicle*, Vehicle*> vehiclesToSwap = randomChooseVehicle(solution);

428 Vehicle& v1 = *obj.first;
429
430 for (auto& location : obj.second) { // LOOP SELECTOR LOCATIONS
431 Location& l1 = *location;
432 if (l1.client==nullptr) { // Skip depot
433 continue;

437 for (auto& path : solution.locationsInSolution) { // Iterate over path for vehicles
